Nested synchronization is error prone, please use explicit lock from java.util.concurrent.locks.Lock

    
    private Map config = new HashMap<>();
    public void update( String key, String value )
    {
        try
        {
            lock.lock();
            if( config.get( key ) != null )
            {
                config.put( key, value );
            }
        }
        finally
        {
            lock.unlock();
        }
    }
